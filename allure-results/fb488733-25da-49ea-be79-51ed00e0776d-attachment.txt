//definicion del contrato json
const schema = {
  "type": "object",
  "required": [
    "statusCode", 
    "statusTitle", 
    "path", 
    "method", 
    "message", 
    "data", 
    "timestamp", 
    "requestId"
    ],
  "properties": {
    "statusCode": { "type": "integer" },
    "statusTitle": { "type": "string" },
    "path": { "type": "string" },
    "method": { "type": "string" },
    "message": {
      "type": "array",
      "items": { "type": "string" }
    },
    "data": {
      "type": "array",
      "items": {
        "type": "object",
        "required": ["taxId", "policyNumber"],
        "properties": {
          "taxId": { "type": "string" },
          "policyNumber": { "type": "string" },
        }
      }
    },
    "timestamp": { "type": "string" },
    "requestId": { "type": "string" }
  }
};

//obtencion de la respuesta en formato json.
const response = pm.response.json();

//Pruebas de validación del contrato json.
pm.test("Formato de response corresponde al esperado",()=>{
    const Ajv = require('ajv');
    const ajv = new Ajv({ allErrors: true});
    const validar = ajv.compile(schema);
    const esValido = validar(response)

    if (!esValido) {
        //se muestran los errores de validación por consola
        console.log("Errores de validación:", validar.errors);

        //se formatean los errores para mostrarlos en el reporte de Postman
        const detalleError = validar.errors.map(err => {
            const path = err.dataPath || "root";
            return `${path}: ${err.message}`});
        pm.expect.fail("Fallo en el esquema: " + detalleError);
    }

    pm.expect(esValido).to.be.true;
});

pm.test("Formato de TaxId 12345678-9 corresponde al esperado", () => {
    for (let i = 0; i < response.data.length; i++) {
        pm.expect(response.data[i].taxId).to.match(/^\d{7,8}-[\dkK]$/);
    }
})

pm.test("Tiempo de respuesta menor o igual a 500ms", ()=> {
    pm.expect(pm.response.responseTime).to.be.below(500);
});

pm.test("Mensaje contiene texto esperado", ()=>{
    regex404 = /No se encontraron registros con los criterios/i
    regex200 = /Solicitud completada con éxito/i
    switch (response.statusCode) {
        case 200:
        pm.expect(response.message).to.match(regex200);
        break;
        case 404:
        pm.expect(response.message).to.match(regex404)
        break;
        default: pm.expect.fail('Error 500 o date vacío o inválido')
    }
})
